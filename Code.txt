CREATE OR REPLACE VIEW OPS_AI_MONITOR.EVENT_LOGS.V_UNIFIED_FAILURES
AS
SELECT
    QUERY_ID                                              AS EVENT_ID,
    'QUERY_REALTIME'                                      AS SOURCE_TYPE,
    ERROR_MESSAGE,
    QUERY_TEXT,
    USER_NAME,
    WAREHOUSE_NAME                                        AS WAREHOUSE,
    DATABASE_NAME                                         AS "DATABASE",
    SCHEMA_NAME                                           AS "SCHEMA",
    START_TIME
FROM TABLE(
    OPS_AI_MONITOR.INFORMATION_SCHEMA.QUERY_HISTORY(
        END_TIME_RANGE_START => DATEADD('MINUTES', -10, CURRENT_TIMESTAMP()),
        END_TIME_RANGE_END   => CURRENT_TIMESTAMP(),
        RESULT_LIMIT         => 10000
    )
)
WHERE EXECUTION_STATUS IN ('FAILED_WITH_ERROR', 'FAILED_WITH_INCIDENT')


UNION ALL


SELECT
    COALESCE(
        QUERY_ID,
        NAME || '_' || TO_VARCHAR(SCHEDULED_TIME, 'YYYYMMDDHH24MISS')
    )                                                     AS EVENT_ID,
    'TASK'                                                AS SOURCE_TYPE,
    ERROR_MESSAGE,
    QUERY_TEXT,
    NULL::VARCHAR                                         AS USER_NAME,
    NULL::VARCHAR                                         AS WAREHOUSE,
    DATABASE_NAME                                         AS "DATABASE",
    SCHEMA_NAME                                           AS "SCHEMA",
    COALESCE(QUERY_START_TIME, SCHEDULED_TIME)            AS START_TIME
FROM TABLE(
    SNOWFLAKE.INFORMATION_SCHEMA.TASK_HISTORY(
        SCHEDULED_TIME_RANGE_START => DATEADD('MINUTES', -10, CURRENT_TIMESTAMP()),
        SCHEDULED_TIME_RANGE_END   => CURRENT_TIMESTAMP(),
        RESULT_LIMIT               => 10000,
        ERROR_ONLY                 => TRUE
    )
)


UNION ALL


SELECT
    QUERY_ID                                              AS EVENT_ID,
    'QUERY_ACCOUNT_USAGE'                                 AS SOURCE_TYPE,
    ERROR_MESSAGE,
    QUERY_TEXT,
    USER_NAME,
    WAREHOUSE_NAME                                        AS WAREHOUSE,
    DATABASE_NAME                                         AS "DATABASE",
    SCHEMA_NAME                                           AS "SCHEMA",
    START_TIME
FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY
WHERE EXECUTION_STATUS IN ('FAIL', 'INCIDENT')
  AND START_TIME >= DATEADD('MINUTES', -10, CURRENT_TIMESTAMP())


UNION ALL


SELECT
    COALESCE(QUERY_ID, NAME || '_DT_' || TO_VARCHAR(DATA_TIMESTAMP, 'YYYYMMDDHH24MISS'))
                                                          AS EVENT_ID,
    'DYNAMIC_TABLE'                                       AS SOURCE_TYPE,
    STATE_MESSAGE                                         AS ERROR_MESSAGE,
    QUALIFIED_NAME                                        AS QUERY_TEXT,
    NULL::VARCHAR                                         AS USER_NAME,
    WAREHOUSE                                             AS WAREHOUSE,
    DATABASE_NAME                                         AS "DATABASE",
    SCHEMA_NAME                                           AS "SCHEMA",
    REFRESH_START_TIME                                    AS START_TIME
FROM TABLE(
    SNOWFLAKE.INFORMATION_SCHEMA.DYNAMIC_TABLE_REFRESH_HISTORY(
        ERROR_ONLY => TRUE
    )
)
WHERE REFRESH_START_TIME >= DATEADD('MINUTES', -10, CURRENT_TIMESTAMP())


UNION ALL


SELECT
    MD5(PIPE_NAME || FILE_NAME || TO_VARCHAR(LAST_LOAD_TIME))
                                                          AS EVENT_ID,
    'SNOWPIPE'                                            AS SOURCE_TYPE,
    FIRST_ERROR_MESSAGE                                   AS ERROR_MESSAGE,
    FILE_NAME                                             AS QUERY_TEXT,
    NULL::VARCHAR                                         AS USER_NAME,
    NULL::VARCHAR                                         AS WAREHOUSE,
    TABLE_CATALOG_NAME                                    AS "DATABASE",
    TABLE_SCHEMA_NAME                                     AS "SCHEMA",
    LAST_LOAD_TIME                                        AS START_TIME
FROM SNOWFLAKE.ACCOUNT_USAGE.COPY_HISTORY
WHERE STATUS IN ('Load failed', 'Partially loaded')
  AND LAST_LOAD_TIME >= DATEADD('HOURS', -2, CURRENT_TIMESTAMP())
  AND PIPE_NAME IS NOT NULL


UNION ALL


SELECT
    'STALE_STREAM_' || STREAM_NAME                        AS EVENT_ID,
    'STREAM_STALE'                                        AS SOURCE_TYPE,
    'Stream is stale: data retention period exceeded'     AS ERROR_MESSAGE,
    STREAM_NAME                                           AS QUERY_TEXT,
    NULL::VARCHAR                                         AS USER_NAME,
    NULL::VARCHAR                                         AS WAREHOUSE,
    DATABASE_NAME                                         AS "DATABASE",
    SCHEMA_NAME                                           AS "SCHEMA",
    DETECTED_AT                                           AS START_TIME
FROM OPS_AI_MONITOR.EVENT_LOGS.STALE_STREAMS
WHERE DETECTED_AT >= DATEADD('MINUTES', -10, CURRENT_TIMESTAMP())


UNION ALL


SELECT
    EVENT_ID,
    SOURCE_SYSTEM                                         AS SOURCE_TYPE,
    ERROR_MESSAGE,
    QUERY_TEXT,
    USER_NAME,
    WAREHOUSE_NAME                                        AS WAREHOUSE,
    DATABASE_NAME                                         AS "DATABASE",
    SCHEMA_NAME                                           AS "SCHEMA",
    START_TIME
FROM OPS_AI_MONITOR.EVENT_LOGS.CUSTOM_LOGS
WHERE STATUS = 'FAILED'
  AND CREATED_AT >= DATEADD('HOURS', -24, CURRENT_TIMESTAMP());

CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_AI_QUEUE
AS
SELECT
    FE.EVENT_ID,
    FE.ERROR_MESSAGE,
    FE.QUERY_TEXT,
    FE.START_TIME
FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS EA
    ON FE.EVENT_ID = EA.EVENT_ID
WHERE EA.EVENT_ID IS NULL
   OR (EA.STATUS = 'FAILED' AND EA.RETRY_COUNT < 3)
ORDER BY FE.START_TIME DESC
LIMIT 20;


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_FAILURE_INSIGHTS
AS
SELECT
    FE.EVENT_ID,

    FE.START_TIME,

    EA.SEVERITY,

    EA.FIX_SQL,
    EA.CONFIDENCE_SCORE,

FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS EA
    ON FE.EVENT_ID = EA.EVENT_ID

-- ============================================================================
-- SECTION 6: VIEWS - CLASSIFICATION & ENRICHMENT
-- ============================================================================


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_FAILURE_INSIGHTS_ENRICHED
AS
SELECT
    FE.EVENT_ID,
    FE.SOURCE_TYPE,
    FE.ERROR_MESSAGE,
    FE.QUERY_TEXT,
    FE.USER_NAME,
    FE.WAREHOUSE_NAME,
    FE.DATABASE_NAME,
    FE.SCHEMA_NAME,
    FE.START_TIME,


    CASE
        WHEN FE.SOURCE_TYPE = 'FIVETRAN'         THEN 'FIVETRAN'
        WHEN FE.SOURCE_TYPE = 'DBT'              THEN 'DBT'
        WHEN FE.SOURCE_TYPE = 'CUSTOM'           THEN 'CUSTOM_ETL'
        WHEN FE.SOURCE_TYPE = 'DYNAMIC_TABLE'    THEN 'DYNAMIC_TABLE'
        WHEN FE.SOURCE_TYPE = 'SNOWPIPE'         THEN 'SNOWPIPE'
        WHEN FE.SOURCE_TYPE IN ('TASK', 'TASK_REALTIME') THEN 'TASK'
        WHEN FE.SOURCE_TYPE = 'STORED_PROC'      THEN 'STORED_PROC'
        WHEN FE.SOURCE_TYPE ILIKE '%ELT%'        THEN 'ELT'
        ELSE 'SQL'
    END AS PIPELINE_TYPE,


    COALESCE(
        CL.PIPELINE_NAME,
        REGEXP_SUBSTR(FE.QUERY_TEXT, 'dbt\\s+run\\s+--select\\s+(\\S+)', 1, 1, 'ie'),
        REGEXP_SUBSTR(FE.QUERY_TEXT, 'TASK\\s+(\\S+)', 1, 1, 'ie'),
        REGEXP_SUBSTR(FE.QUERY_TEXT, 'PROCEDURE\\s+(\\S+)', 1, 1, 'ie'),
        REGEXP_SUBSTR(FE.QUERY_TEXT, 'CALL\\s+(\\S+)', 1, 1, 'ie'),
        REGEXP_SUBSTR(FE.ERROR_MESSAGE, 'Task\\s+(\\S+)', 1, 1, 'ie'),
        REGEXP_SUBSTR(FE.ERROR_MESSAGE, 'table\\s+''?([\\w\\.]+)''?', 1, 1, 'ie'),
        REGEXP_SUBSTR(FE.ERROR_MESSAGE, 'Object\\s+''?([\\w\\.]+)''?', 1, 1, 'ie'),
        CASE
            WHEN FE.DATABASE_NAME IS NOT NULL AND FE.SCHEMA_NAME IS NOT NULL
            THEN FE.DATABASE_NAME || '.' || FE.SCHEMA_NAME
            ELSE LEFT(COALESCE(FE.QUERY_TEXT, 'Unknown'), 50)
        END
    ) AS PIPELINE_NAME,


    CASE
        WHEN FE.ERROR_MESSAGE ILIKE '%does not exist%' OR FE.ERROR_MESSAGE ILIKE '%not found%' THEN 'MISSING_OBJECT'
        WHEN FE.ERROR_MESSAGE ILIKE '%not authorized%' OR FE.ERROR_MESSAGE ILIKE '%insufficient privileges%' THEN 'PERMISSION'
        WHEN FE.ERROR_MESSAGE ILIKE '%syntax error%' OR FE.ERROR_MESSAGE ILIKE '%compilation error%' THEN 'SYNTAX'
        WHEN FE.ERROR_MESSAGE ILIKE '%division by zero%' THEN 'RUNTIME'
        WHEN FE.ERROR_MESSAGE ILIKE '%stale%' THEN 'STALENESS'
        WHEN FE.ERROR_MESSAGE ILIKE '%file format%' OR FE.ERROR_MESSAGE ILIKE '%copy into%' THEN 'DATA_LOAD'
        WHEN FE.ERROR_MESSAGE ILIKE '%connection%' OR FE.ERROR_MESSAGE ILIKE '%timeout%' OR FE.ERROR_MESSAGE ILIKE '%rate limit%' THEN 'CONNECTIVITY'
        WHEN FE.ERROR_MESSAGE ILIKE '%type mismatch%' OR FE.ERROR_MESSAGE ILIKE '%numeric%' OR FE.ERROR_MESSAGE ILIKE '%cast%' THEN 'DATA_TYPE'
        ELSE 'OTHER'
    END AS ERROR_CATEGORY,


    EA.ROOT_CAUSE,
    COALESCE(EA.SEVERITY, 'MEDIUM') AS SEVERITY,
    EA.SUGGESTED_FIX,
    EA.FIX_SQL,
    EA.TEST_SQL,
    EA.CONFIDENCE_SCORE,


    CASE
        WHEN EA.STATUS IS NULL THEN 'PENDING'
        WHEN EA.STATUS = 'SUCCESS' THEN 'ANALYZED'
        WHEN EA.STATUS = 'FAILED' THEN 'ANALYSIS_FAILED'
        ELSE EA.STATUS
    END AS AI_STATUS,


    FAQ.APPROVAL_STATUS AS FIX_STATUS,
    AFL.EXECUTION_STATUS AS FIX_EXECUTION,
    EA.STATUS AS _RAW_AI_STATUS


FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS EA ON FE.EVENT_ID = EA.EVENT_ID
LEFT JOIN OPS_AI_MONITOR.EVENT_LOGS.CUSTOM_LOGS CL ON FE.EVENT_ID = CL.EVENT_ID
LEFT JOIN (
    SELECT EVENT_ID, APPROVAL_STATUS,
           ROW_NUMBER() OVER (PARTITION BY EVENT_ID ORDER BY CREATED_AT DESC) AS RN
    FROM OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
) FAQ ON FE.EVENT_ID = FAQ.EVENT_ID AND FAQ.RN = 1
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.AUTO_FIX_LOGS AFL ON FE.EVENT_ID = AFL.EVENT_ID
ORDER BY FE.START_TIME DESC;

CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_FAILURE_METRICS
AS
SELECT
    COUNT(*)                                                              AS TOTAL_FAILURES,
    COUNT_IF(FE.SOURCE_TYPE IN ('QUERY_REALTIME', 'QUERY_ACCOUNT_USAGE')) AS QUERY_FAILURES,
    COUNT_IF(FE.SOURCE_TYPE = 'TASK')                                     AS TASK_FAILURES,
    COUNT_IF(FE.SOURCE_TYPE = 'DYNAMIC_TABLE')                            AS DYNAMIC_TABLE_FAILURES,
    COUNT_IF(FE.SOURCE_TYPE = 'SNOWPIPE')                                 AS SNOWPIPE_FAILURES,
    COUNT_IF(FE.SOURCE_TYPE = 'STREAM_STALE')                             AS STREAM_FAILURES,
    COUNT_IF(FE.SOURCE_TYPE IN ('FIVETRAN', 'DBT', 'CUSTOM'))            AS EXTERNAL_FAILURES,
    COUNT_IF(FE.ERROR_MESSAGE ILIKE '%does not exist%'
          OR FE.ERROR_MESSAGE ILIKE '%not authorized%')                   AS MISSING_OBJECT_COUNT,
    COUNT_IF(EA.SEVERITY = 'HIGH')                                        AS HIGH_SEVERITY_COUNT,
    COUNT_IF(EA.SEVERITY = 'MEDIUM')                                      AS MEDIUM_SEVERITY_COUNT,
    COUNT_IF(EA.SEVERITY = 'LOW')                                         AS LOW_SEVERITY_COUNT,
    COUNT_IF(EA.STATUS = 'SUCCESS')                                       AS AI_SUCCESS_COUNT,
    COUNT_IF(EA.STATUS = 'FAILED')                                        AS AI_FAILED_COUNT,
    COUNT_IF(EA.EVENT_ID IS NULL)                                         AS AI_PENDING_COUNT,
    ROUND(COUNT_IF(EA.STATUS = 'SUCCESS') * 100.0 / NULLIF(COUNT(*), 0), 1) AS AI_SUCCESS_RATE_PCT,
    MIN(FE.START_TIME)                                                    AS EARLIEST_FAILURE,
    MAX(FE.START_TIME)                                                    AS LATEST_FAILURE
FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS EA
    ON FE.EVENT_ID = EA.EVENT_ID
WHERE FE.START_TIME >= DATEADD('DAY', -1, CURRENT_TIMESTAMP());


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_SLA_BREACH
AS
SELECT
    FE.EVENT_ID,
    VE.PIPELINE_TYPE,
    FE.SOURCE_TYPE,
    ROUND(QH.TOTAL_ELAPSED_TIME / 1000, 1)               AS EXECUTION_TIME_SECONDS,
    CASE
        WHEN QH.TOTAL_ELAPSED_TIME / 1000 > 300 THEN 'BREACHED'
        ELSE 'OK'
    END                                                    AS SLA_STATUS,
    CASE
        WHEN QH.TOTAL_ELAPSED_TIME / 1000 > 600 THEN 'CRITICAL: exceeded 2x SLA (>10 min)'
        WHEN QH.TOTAL_ELAPSED_TIME / 1000 > 300 THEN 'WARNING: exceeded SLA threshold (>5 min)'
        ELSE 'Within SLA'
    END                                                    AS BREACH_REASON,
    VE.ERROR_CATEGORY,
    VE.SEVERITY,
    FE.START_TIME
FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
INNER JOIN SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY QH
    ON FE.EVENT_ID = QH.QUERY_ID
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.V_FAILURE_INSIGHTS_ENRICHED VE
    ON FE.EVENT_ID = VE.EVENT_ID
WHERE FE.START_TIME >= DATEADD('DAY', -1, CURRENT_TIMESTAMP())
ORDER BY EXECUTION_TIME_SECONDS DESC
UNION ALL
SELECT EVENT_ID, PIPELINE_TYPE, SOURCE_TYPE, EXECUTION_TIME_SECONDS,
       SLA_STATUS, BREACH_REASON, ERROR_CATEGORY, SEVERITY, START_TIME
FROM OPS_AI_MONITOR.AI_ENGINE.SLA_DEMO_BREACHES;


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_SLA_SUMMARY
AS
SELECT
    PIPELINE_TYPE,
    COUNT(*)                                                             AS TOTAL_RUNS,
    COUNT_IF(SLA_STATUS = 'BREACHED')                                    AS BREACHED_RUNS,
    ROUND((COUNT(*) - COUNT_IF(SLA_STATUS = 'BREACHED')) * 100.0
          / NULLIF(COUNT(*), 0), 1)                                      AS SUCCESS_RATE_PCT,
    ROUND(AVG(EXECUTION_TIME_SECONDS), 2)                                AS AVG_EXECUTION_TIME_SEC,
    ROUND(MAX(EXECUTION_TIME_SECONDS), 2)                                AS MAX_EXECUTION_TIME_SEC,
    ROUND(MEDIAN(EXECUTION_TIME_SECONDS), 2)                             AS P50_EXECUTION_TIME_SEC
FROM OPS_AI_MONITOR.AI_ENGINE.V_SLA_BREACH
GROUP BY PIPELINE_TYPE
ORDER BY BREACHED_RUNS DESC, TOTAL_RUNS DESC;


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_CRITICAL_ALERTS
AS
WITH REPEAT_OFFENDERS AS (
    SELECT
        EVENT_ID,
        ERROR_MESSAGE,
        COUNT(*) OVER (
            PARTITION BY ERROR_MESSAGE
            ORDER BY START_TIME
            RANGE BETWEEN INTERVAL '10 MINUTES' PRECEDING AND CURRENT ROW
        ) AS FAILURES_IN_WINDOW
    FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS
    WHERE START_TIME >= DATEADD('DAY', -1, CURRENT_TIMESTAMP())
),
SLA AS (
    SELECT EVENT_ID, SLA_STATUS
    FROM OPS_AI_MONITOR.AI_ENGINE.V_SLA_BREACH
    WHERE SLA_STATUS = 'BREACHED'
)
SELECT
    FE.EVENT_ID,
    VE.PIPELINE_TYPE,
    VE.ERROR_CATEGORY,
    VE.SEVERITY,
    CASE
        WHEN VE.SEVERITY = 'HIGH' AND SLA.EVENT_ID IS NOT NULL AND RO.FAILURES_IN_WINDOW > 3
            THEN 'HIGH_SEV + SLA_BREACH + REPEATED'
        WHEN VE.SEVERITY = 'HIGH' AND SLA.EVENT_ID IS NOT NULL
            THEN 'HIGH_SEV + SLA_BREACH'
        WHEN VE.SEVERITY = 'HIGH' AND RO.FAILURES_IN_WINDOW > 3
            THEN 'HIGH_SEV + REPEATED'
        WHEN SLA.EVENT_ID IS NOT NULL AND RO.FAILURES_IN_WINDOW > 3
            THEN 'SLA_BREACH + REPEATED'
        WHEN VE.SEVERITY = 'HIGH'
            THEN 'HIGH_SEVERITY'
        WHEN SLA.EVENT_ID IS NOT NULL
            THEN 'SLA_BREACH'
        WHEN RO.FAILURES_IN_WINDOW > 3
            THEN 'REPEATED_FAILURE'
    END AS ISSUE_TYPE,
    CASE
        WHEN VE.SEVERITY = 'HIGH' AND SLA.EVENT_ID IS NOT NULL  THEN 'P1'
        WHEN VE.SEVERITY = 'HIGH' OR SLA.EVENT_ID IS NOT NULL   THEN 'P2'
        WHEN RO.FAILURES_IN_WINDOW > 3                          THEN 'P3'
    END AS ALERT_FLAG,
    RO.FAILURES_IN_WINDOW,
    FE.ERROR_MESSAGE,
    FE.START_TIME
FROM OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
JOIN OPS_AI_MONITOR.AI_ENGINE.V_FAILURE_INSIGHTS_ENRICHED VE
    ON FE.EVENT_ID = VE.EVENT_ID
LEFT JOIN SLA
    ON FE.EVENT_ID = SLA.EVENT_ID
LEFT JOIN REPEAT_OFFENDERS RO
    ON FE.EVENT_ID = RO.EVENT_ID
WHERE FE.START_TIME >= DATEADD('DAY', -1, CURRENT_TIMESTAMP())
  AND (VE.SEVERITY = 'HIGH'
       OR SLA.EVENT_ID IS NOT NULL
       OR RO.FAILURES_IN_WINDOW > 3)
ORDER BY
    CASE ALERT_FLAG WHEN 'P1' THEN 1 WHEN 'P2' THEN 2 WHEN 'P3' THEN 3 END,
    FE.START_TIME DESC;

CREATE OR REPLACE TASK OPS_AI_MONITOR.EVENT_LOGS.TSK_AI_ANALYZE
    WAREHOUSE = OPS_MONITOR_WH
    COMMENT = 'Runs KB-enriched Cortex COMPLETE on queued failures via V_AI_QUEUE, 10 per run. fix_sql is ALWAYS guaranteed.'
    AFTER OPS_AI_MONITOR.EVENT_LOGS.TSK_INGEST_FAILURES
AS
    MERGE INTO OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS AS TGT
    USING (
        WITH QUEUED AS (
            SELECT EVENT_ID, ERROR_MESSAGE, QUERY_TEXT
            FROM OPS_AI_MONITOR.AI_ENGINE.V_AI_QUEUE
            LIMIT 10
        ),
        KB_ENRICHED AS (
            SELECT
                Q.EVENT_ID,
                Q.ERROR_MESSAGE,
                Q.QUERY_TEXT,
                KB.RECOMMENDED_FIX AS KB_HINT,
                KB.FIX_SQL         AS KB_FIX_SQL
            FROM QUEUED Q
            LEFT JOIN OPS_AI_MONITOR.METADATA.ERROR_KB KB
                ON Q.ERROR_MESSAGE ILIKE '%' || KB.PATTERN || '%'
            QUALIFY ROW_NUMBER() OVER (
                PARTITION BY Q.EVENT_ID
                ORDER BY LENGTH(KB.PATTERN) DESC NULLS LAST
            ) = 1
        ),
        AI_RAW AS (
            SELECT
                EVENT_ID,
                ERROR_MESSAGE,
                QUERY_TEXT,
                SNOWFLAKE.CORTEX.COMPLETE(
                    'mistral-large2',
                    CONCAT(
                        'You are a Snowflake SQL expert analyzing pipeline failures.\n',
                        'Determine root cause, provide an actionable fix, AND a validation test query.\n\n',
                        'CRITICAL RULES:\n',
                        '- fix_sql MUST ALWAYS contain executable Snowflake SQL. NEVER return null or empty fix_sql.\n',
                        '- test_sql MUST ALWAYS contain a safe read-only SELECT query to validate the fix worked.\n\n',
                        'If you know the exact fix, provide it.\n',
                        'If you are unsure, generate a SAFE diagnostic SQL instead using these rules:\n',
                        '- Missing table/object → SHOW TABLES LIKE ''<object_name>'' IN ACCOUNT\n',
                        '- Permission/privilege error → SHOW GRANTS ON <object_type> <object_name>\n',
                        '- Syntax/compilation error → SELECT ''SYNTAX_ERROR'' AS issue, ''<brief_description>'' AS detail\n',
                        '- Data type/cast error → SELECT TRY_CAST(''<value>'' AS <target_type>) AS diagnostic\n',
                        '- Division by zero → SELECT ''DIVISION_BY_ZERO'' AS issue, ''Add NULLIF guard'' AS fix\n',
                        '- Stale stream → SHOW STREAMS LIKE ''<stream_name>'' IN ACCOUNT\n',
                        '- Load/file format error → SHOW STAGES LIKE ''<stage_name>'' IN ACCOUNT\n',
                        '- Warehouse issue → SHOW WAREHOUSES LIKE ''<warehouse_name>''\n',
                        '- Unknown/other → SELECT ''<first_100_chars_of_error>'' AS debug_info\n\n',
                        'FORBIDDEN in fix_sql: DROP, DELETE, TRUNCATE, CREATE OR REPLACE. These are dangerous.\n',
                        'FORBIDDEN in test_sql: Any DML/DDL (INSERT, UPDATE, DELETE, DROP, CREATE, ALTER, GRANT, MERGE).\n\n',
                        'test_sql RULES (for validating the fix):\n',
                        '- MUST be a safe read-only SELECT. No DML/DDL.\n',
                        '- Missing table fix → SELECT COUNT(*) FROM <table_name>\n',
                        '- Permission fix → SELECT CURRENT_ROLE() AS role_check\n',
                        '- Syntax fix → Re-run the corrected query (SELECT only)\n',
                        '- Object creation fix → SELECT COUNT(*) FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = ''<name>''\n',
                        '- Data type fix → SELECT TRY_CAST(''<value>'' AS <type>) AS cast_check\n',
                        '- If unsure → SELECT 1 AS VALIDATION_OK\n\n',
                        'Common failure categories: missing object, insufficient privileges, ',
                        'SQL syntax error, division by zero, data type mismatch, resource contention, ',
                        'stale stream, load failure, schema drift, connectivity.\n\n',
                        'ERROR: ', COALESCE(LEFT(ERROR_MESSAGE, 1000), 'N/A'), '\n\n',
                        'QUERY: ', COALESCE(LEFT(QUERY_TEXT, 2000), 'N/A'), '\n\n',
                        IFF(KB_HINT IS NOT NULL,
                            CONCAT('Hint from knowledge base: ', KB_HINT, '\n',
                                   'Reference fix: ', COALESCE(KB_FIX_SQL, 'N/A'), '\n\n'),
                            ''),
                        'Return ONLY a valid JSON object with these exact keys:\n',
                        '{"root_cause":"...","severity":"LOW|MEDIUM|HIGH",',
                        '"suggested_fix":"...","fix_sql":"<MUST BE EXECUTABLE SQL>",',
                        '"test_sql":"<SAFE SELECT TO VALIDATE FIX>","confidence_score":0.0}\n\n',
                        'Rules:\n',
                        '- fix_sql MUST be present and contain valid executable Snowflake SQL. NEVER empty. NEVER null.\n',
                        '- test_sql MUST be present and contain a safe SELECT query. NEVER empty. NEVER null.\n',
                        '- fix_sql must NOT contain DROP, DELETE, TRUNCATE, or CREATE OR REPLACE.\n',
                        '- test_sql must be read-only SELECT only.\n',
                        '- confidence_score between 0.0 and 1.0\n',
                        '- No markdown. No explanation. JSON only.'
                    )
                ) AS RAW_RESPONSE
            FROM KB_ENRICHED
        ),
        AI_PARSED AS (
            SELECT
                EVENT_ID,
                ERROR_MESSAGE,
                QUERY_TEXT,
                RAW_RESPONSE,
                TRY_PARSE_JSON(
                    TRIM(REGEXP_REPLACE(RAW_RESPONSE, '```[a-zA-Z]*|```', ''))
                ) AS PARSED
            FROM AI_RAW
        ),
        AI_VALIDATED AS (
            SELECT
                EVENT_ID,
                ERROR_MESSAGE,
                QUERY_TEXT,
                RAW_RESPONSE,
                PARSED,
                PARSED:root_cause::VARCHAR       AS RAW_ROOT_CAUSE,
                PARSED:severity::VARCHAR         AS RAW_SEVERITY,
                PARSED:suggested_fix::VARCHAR    AS RAW_SUGGESTED_FIX,
                PARSED:fix_sql::VARCHAR          AS RAW_FIX_SQL,
                PARSED:test_sql::VARCHAR         AS RAW_TEST_SQL,
                PARSED:confidence_score::FLOAT   AS RAW_CONFIDENCE
            FROM AI_PARSED
        ),
        AI_SAFE AS (
            SELECT
                EVENT_ID,
                RAW_RESPONSE,
                PARSED,
                COALESCE(RAW_ROOT_CAUSE, 'Unable to determine root cause') AS ROOT_CAUSE,
                COALESCE(RAW_SEVERITY, 'MEDIUM')                           AS SEVERITY,
                COALESCE(RAW_SUGGESTED_FIX, 'Review error details and investigate manually') AS SUGGESTED_FIX,
                COALESCE(RAW_CONFIDENCE, 0.3)                              AS CONFIDENCE_SCORE,
                CASE
                    WHEN RAW_FIX_SQL IS NOT NULL
                         AND TRIM(RAW_FIX_SQL) != ''
                         AND UPPER(RAW_FIX_SQL) NOT LIKE '%DROP %'
                         AND UPPER(RAW_FIX_SQL) NOT LIKE '%DELETE %'
                         AND UPPER(RAW_FIX_SQL) NOT LIKE '%TRUNCATE %'
                         AND UPPER(RAW_FIX_SQL) NOT LIKE '%CREATE OR REPLACE %'
                        THEN RAW_FIX_SQL
                    WHEN ERROR_MESSAGE ILIKE '%does not exist%'
                      OR ERROR_MESSAGE ILIKE '%object%not found%'
                      OR ERROR_MESSAGE ILIKE '%unknown%table%'
                        THEN CONCAT('SHOW TABLES LIKE ''%',
                             COALESCE(
                                 REGEXP_SUBSTR(ERROR_MESSAGE, '''([^'']+)''', 1, 1, 'e'),
                                 'UNKNOWN'
                             ), '%'' IN ACCOUNT')
                    WHEN ERROR_MESSAGE ILIKE '%insufficient privileges%'
                      OR ERROR_MESSAGE ILIKE '%access denied%'
                      OR ERROR_MESSAGE ILIKE '%permission%denied%'
                      OR ERROR_MESSAGE ILIKE '%not authorized%'
                        THEN 'SHOW GRANTS TO ROLE ' || CURRENT_ROLE()
                    WHEN ERROR_MESSAGE ILIKE '%SQL compilation error%'
                      OR ERROR_MESSAGE ILIKE '%syntax error%'
                      OR ERROR_MESSAGE ILIKE '%parse error%'
                        THEN CONCAT('SELECT ''SYNTAX_ERROR'' AS issue, ',
                             '''', REPLACE(LEFT(COALESCE(ERROR_MESSAGE, 'Unknown syntax error'), 200), '''', '\\'''), ''' AS error_detail')
                    WHEN ERROR_MESSAGE ILIKE '%division by zero%'
                        THEN 'SELECT ''DIVISION_BY_ZERO'' AS issue, ''Wrap divisor with NULLIF(col, 0)'' AS suggested_fix'
                    WHEN ERROR_MESSAGE ILIKE '%stale%'
                      OR ERROR_MESSAGE ILIKE '%retention%'
                        THEN CONCAT('SHOW STREAMS LIKE ''%',
                             COALESCE(
                                 REGEXP_SUBSTR(ERROR_MESSAGE, '''([^'']+)''', 1, 1, 'e'),
                                 REGEXP_SUBSTR(QUERY_TEXT, '''([^'']+)''', 1, 1, 'e')
                             ), '%'' IN ACCOUNT')
                    WHEN ERROR_MESSAGE ILIKE '%load failed%'
                      OR ERROR_MESSAGE ILIKE '%file format%'
                      OR ERROR_MESSAGE ILIKE '%field delimiter%'
                        THEN 'SHOW STAGES IN ACCOUNT'
                    WHEN ERROR_MESSAGE ILIKE '%warehouse%'
                        THEN 'SHOW WAREHOUSES'
                    WHEN ERROR_MESSAGE ILIKE '%numeric value%'
                      OR ERROR_MESSAGE ILIKE '%out of range%'
                      OR ERROR_MESSAGE ILIKE '%type mismatch%'
                        THEN CONCAT('SELECT ''DATA_TYPE_ERROR'' AS issue, ',
                             '''', REPLACE(LEFT(COALESCE(ERROR_MESSAGE, 'Unknown type error'), 200), '''', '\\'''), ''' AS error_detail')
                    WHEN ERROR_MESSAGE ILIKE '%connection%'
                      OR ERROR_MESSAGE ILIKE '%SSL%'
                      OR ERROR_MESSAGE ILIKE '%rate limit%'
                      OR ERROR_MESSAGE ILIKE '%HTTP 429%'
                        THEN CONCAT('SELECT ''CONNECTIVITY_ERROR'' AS issue, ',
                             '''', REPLACE(LEFT(COALESCE(ERROR_MESSAGE, 'Unknown connectivity error'), 200), '''', '\\'''), ''' AS error_detail')
                    ELSE CONCAT('SELECT ',
                         '''', REPLACE(LEFT(COALESCE(ERROR_MESSAGE, 'Unknown error'), 200), '''', '\\'''), ''' AS debug_info')
                END AS FIX_SQL,
                CASE
                    WHEN RAW_TEST_SQL IS NOT NULL
                         AND TRIM(RAW_TEST_SQL) != ''
                         AND UPPER(LEFT(TRIM(RAW_TEST_SQL), 6)) = 'SELECT'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%DROP %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%DELETE %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%INSERT %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%UPDATE %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%MERGE %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%CREATE %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%ALTER %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%GRANT %'
                         AND UPPER(RAW_TEST_SQL) NOT LIKE '%TRUNCATE %'
                        THEN RAW_TEST_SQL
                    ELSE 'SELECT 1 AS VALIDATION_OK'
                END AS TEST_SQL
            FROM AI_VALIDATED
        )
        SELECT
            EVENT_ID,
            ROOT_CAUSE,
            SEVERITY,
            SUGGESTED_FIX,
            FIX_SQL,
            TEST_SQL,
            CONFIDENCE_SCORE,
            COALESCE(PARSED, OBJECT_CONSTRUCT(
                'error', 'PARSE_FAILED',
                'raw', RAW_RESPONSE
            ))                                        AS RESPONSE_VARIANT,
            IFF(PARSED IS NOT NULL, 'SUCCESS', 'FAILED') AS STATUS
        FROM AI_SAFE
    ) AS SRC
    ON TGT.EVENT_ID = SRC.EVENT_ID
    WHEN MATCHED THEN UPDATE SET
        TGT.ROOT_CAUSE       = SRC.ROOT_CAUSE,
        TGT.SEVERITY         = SRC.SEVERITY,
        TGT.SUGGESTED_FIX    = SRC.SUGGESTED_FIX,
        TGT.FIX_SQL          = SRC.FIX_SQL,
        TGT.TEST_SQL         = SRC.TEST_SQL,
        TGT.CONFIDENCE_SCORE = SRC.CONFIDENCE_SCORE,
        TGT.RAW_RESPONSE     = SRC.RESPONSE_VARIANT,
        TGT.STATUS           = SRC.STATUS,
        TGT.RETRY_COUNT      = TGT.RETRY_COUNT + 1,
        TGT.UPDATED_AT       = CURRENT_TIMESTAMP()
    WHEN NOT MATCHED THEN INSERT (
        EVENT_ID, ROOT_CAUSE, SEVERITY, SUGGESTED_FIX, FIX_SQL, TEST_SQL,
        CONFIDENCE_SCORE, RAW_RESPONSE, STATUS
    )
    VALUES (
        SRC.EVENT_ID, SRC.ROOT_CAUSE, SRC.SEVERITY, SRC.SUGGESTED_FIX, SRC.FIX_SQL, SRC.TEST_SQL,
        SRC.CONFIDENCE_SCORE, SRC.RESPONSE_VARIANT, SRC.STATUS
    );


CREATE OR REPLACE TASK OPS_AI_MONITOR.EVENT_LOGS.TSK_SEND_ALERTS
    WAREHOUSE = OPS_MONITOR_WH
    COMMENT = 'Sends email + Slack alerts for HIGH severity failures with approval workflow instructions'
    AFTER OPS_AI_MONITOR.EVENT_LOGS.TSK_AI_ANALYZE
AS
DECLARE
    ALERT_COUNT   INT;
    PENDING_COUNT INT;
    ALERT_BODY    VARCHAR;
    SLACK_BLOCKS  VARCHAR;
    ALERT_DETAIL  VARCHAR;
    RECENT_ALERTS INT;
BEGIN
    SELECT COUNT(*) INTO :RECENT_ALERTS
    FROM OPS_AI_MONITOR.AI_ENGINE.ALERT_LOG
    WHERE ALERTED_AT >= DATEADD('MINUTE', -5, CURRENT_TIMESTAMP());


    IF (:RECENT_ALERTS >= 10) THEN
        RETURN 'Cooldown: ' || :RECENT_ALERTS || ' alerts sent in last 5 min, skipping';

    SELECT
        CA.EVENT_ID, CA.ALERT_FLAG, CA.ISSUE_TYPE, CA.PIPELINE_TYPE, CA.SEVERITY,

        COALESCE(EA.CONFIDENCE_SCORE, 0) AS CONFIDENCE_SCORE,
        CA.START_TIME,
        FAQ.APPROVAL_STATUS AS FIX_APPROVAL_STATUS,
        FAQ.APPROVAL_POLICY AS FIX_POLICY,

            '<div style="background:#d32f2f;color:#fff;padding:20px 24px;">',
            '<h2 style="margin:0;">&#9888; CoCo Pipeline Alert</h2>',
            '<p style="margin:4px 0 0;">', :ALERT_COUNT, ' new HIGH severity failure(s) | ',
            :PENDING_COUNT, ' fix(es) awaiting approval</p>',
            '</div>',
            '<div style="background:#fff3e0;padding:12px 24px;border-bottom:1px solid #ffe0b2;">',
            '<strong style="color:#e65100;">&#128227; Action Required:</strong> ',
            'Fixes will NOT execute until approved. ',
            '<a href="https://app.snowflake.com/gqb30339/#/streamlit-apps/a753tuigfbe34dmmok3r" ',
            'style="color:#1565c0;font-weight:bold;">Open Dashboard &rarr; Fix Approvals tab</a>',
            '</div>',
            '<div style="padding:20px 24px;">',
            LISTAGG(
                CONCAT(
                    '<div style="border:1px solid #e0e0e0;border-radius:6px;padding:16px;margin-bottom:16px;background:#fafafa;">',
                    '<div style="display:flex;justify-content:space-between;margin-bottom:8px;">',

-- === HIL: ROUTE FIXES TO APPROVAL QUEUE ===
CREATE OR REPLACE TASK OPS_AI_MONITOR.EVENT_LOGS.TSK_ROUTE_FIXES
    WAREHOUSE = OPS_MONITOR_WH
    COMMENT = 'Routes AI-generated fixes to approval queue. HIGH=manual, MEDIUM=manual if confidence<0.9, LOW=auto-approve'
    AFTER OPS_AI_MONITOR.EVENT_LOGS.TSK_AI_ANALYZE
AS
    MERGE INTO OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE AS TGT
    USING (
        SELECT
            EA.EVENT_ID,
            FE.SOURCE_TYPE AS PIPELINE_TYPE,
            EA.SEVERITY,
            EA.CONFIDENCE_SCORE,
            EA.ROOT_CAUSE,
            EA.SUGGESTED_FIX,
            EA.FIX_SQL,
            CASE
                WHEN EA.SEVERITY = 'HIGH' THEN 'MANUAL'
                WHEN EA.SEVERITY = 'MEDIUM' AND EA.CONFIDENCE_SCORE < 0.9 THEN 'MANUAL'
                WHEN EA.SEVERITY = 'MEDIUM' AND EA.CONFIDENCE_SCORE >= 0.9 THEN 'AUTO'
                WHEN EA.SEVERITY = 'LOW' AND EA.CONFIDENCE_SCORE >= 0.8 THEN 'AUTO'
                ELSE 'MANUAL'
            END AS POLICY,
            CASE
                WHEN EA.SEVERITY = 'LOW' AND EA.CONFIDENCE_SCORE >= 0.8 THEN 'AUTO_APPROVED'
                WHEN EA.SEVERITY = 'MEDIUM' AND EA.CONFIDENCE_SCORE >= 0.9 THEN 'AUTO_APPROVED'
                ELSE 'PENDING'
            END AS INITIAL_STATUS
        FROM OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS EA
        JOIN OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
            ON EA.EVENT_ID = FE.EVENT_ID
        WHERE EA.STATUS = 'SUCCESS'
          AND EA.FIX_SQL IS NOT NULL
          AND TRIM(EA.FIX_SQL) != ''
          AND UPPER(EA.FIX_SQL) NOT LIKE '%DROP %'
          AND UPPER(EA.FIX_SQL) NOT LIKE '%DELETE %'
          AND UPPER(EA.FIX_SQL) NOT LIKE '%TRUNCATE %'
          AND EA.CREATED_AT >= DATEADD('HOUR', -2, CURRENT_TIMESTAMP())
    ) AS SRC
        ON TGT.EVENT_ID = SRC.EVENT_ID
    WHEN NOT MATCHED THEN INSERT (
        EVENT_ID, PIPELINE_TYPE, SEVERITY, CONFIDENCE_SCORE,
        ROOT_CAUSE, SUGGESTED_FIX, FIX_SQL,
        APPROVAL_STATUS, APPROVAL_POLICY,
        APPROVED_BY, APPROVED_AT
    ) VALUES (
        SRC.EVENT_ID, SRC.PIPELINE_TYPE, SRC.SEVERITY, SRC.CONFIDENCE_SCORE,
        SRC.ROOT_CAUSE, SRC.SUGGESTED_FIX, SRC.FIX_SQL,
        SRC.INITIAL_STATUS, SRC.POLICY,
        CASE WHEN SRC.INITIAL_STATUS = 'AUTO_APPROVED' THEN 'SYSTEM' ELSE NULL END,
        CASE WHEN SRC.INITIAL_STATUS = 'AUTO_APPROVED' THEN CURRENT_TIMESTAMP() ELSE NULL END
    );


-- === HIL: AUTO-FIX (ONLY APPROVED) ===
CREATE OR REPLACE TASK OPS_AI_MONITOR.EVENT_LOGS.TSK_AUTO_FIX
    WAREHOUSE = OPS_MONITOR_WH
    COMMENT = 'Executes ONLY approved fixes from FIX_APPROVAL_QUEUE (human-approved or auto-approved)'
    AFTER OPS_AI_MONITOR.EVENT_LOGS.TSK_ROUTE_FIXES
AS
DECLARE
    C1 CURSOR FOR
        SELECT FAQ.APPROVAL_ID, FAQ.EVENT_ID, FAQ.FIX_SQL
        FROM OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE FAQ
        LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.AUTO_FIX_LOGS AFL
            ON FAQ.EVENT_ID = AFL.EVENT_ID
        WHERE FAQ.APPROVAL_STATUS IN ('APPROVED', 'AUTO_APPROVED')
          AND FAQ.EXECUTED = FALSE
          AND AFL.EVENT_ID IS NULL
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%DROP %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%DELETE %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%TRUNCATE %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%INSERT %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%UPDATE %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%MERGE %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%CREATE OR REPLACE %'
          AND UPPER(FAQ.FIX_SQL) NOT LIKE '%REVOKE %'
        ORDER BY FAQ.APPROVED_AT ASC
        LIMIT 5;
    V_APPROVAL_ID VARCHAR;
    V_EVENT_ID    VARCHAR;
    V_FIX_SQL     VARCHAR;
BEGIN
    OPEN C1;
    FOR REC IN C1 DO
        V_APPROVAL_ID := REC.APPROVAL_ID;
        V_EVENT_ID    := REC.EVENT_ID;
        V_FIX_SQL     := REC.FIX_SQL;
        BEGIN
            EXECUTE IMMEDIATE :V_FIX_SQL;


            INSERT INTO OPS_AI_MONITOR.AI_ENGINE.AUTO_FIX_LOGS
                (EVENT_ID, FIX_SQL, EXECUTION_STATUS)
            VALUES (:V_EVENT_ID, :V_FIX_SQL, 'SUCCESS');


            UPDATE OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
            SET EXECUTED = TRUE
            WHERE APPROVAL_ID = :V_APPROVAL_ID;
        EXCEPTION
            WHEN OTHER THEN
                INSERT INTO OPS_AI_MONITOR.AI_ENGINE.AUTO_FIX_LOGS
                    (EVENT_ID, FIX_SQL, EXECUTION_STATUS, ERROR_MESSAGE)
                VALUES (:V_EVENT_ID, :V_FIX_SQL, 'FAILED', SQLERRM);


                UPDATE OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
                SET EXECUTED = TRUE
                WHERE APPROVAL_ID = :V_APPROVAL_ID;
        END;
    END FOR;
    CLOSE C1;
END;


-- === HIL: APPROVAL PROCEDURES ===
CREATE OR REPLACE PROCEDURE OPS_AI_MONITOR.AI_ENGINE.SP_APPROVE_FIX(P_EVENT_ID VARCHAR)
RETURNS VARCHAR
LANGUAGE SQL
AS
DECLARE
    V_COUNT INT;
BEGIN
    SELECT COUNT(*) INTO :V_COUNT
    FROM OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
    WHERE EVENT_ID = :P_EVENT_ID AND APPROVAL_STATUS = 'PENDING';
    IF (:V_COUNT = 0) THEN
        RETURN 'No pending fix found for event: ' || :P_EVENT_ID;
    END IF;
    UPDATE OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
    SET APPROVAL_STATUS = 'APPROVED',
        APPROVED_BY = CURRENT_USER(),
        APPROVED_AT = CURRENT_TIMESTAMP()
    WHERE EVENT_ID = :P_EVENT_ID AND APPROVAL_STATUS = 'PENDING';
    RETURN 'APPROVED by ' || CURRENT_USER() || ' — fix will execute on next TSK_AUTO_FIX run';
END;


CREATE OR REPLACE PROCEDURE OPS_AI_MONITOR.AI_ENGINE.SP_REJECT_FIX(P_EVENT_ID VARCHAR, P_REASON VARCHAR)
RETURNS VARCHAR
LANGUAGE SQL
AS
DECLARE
    V_COUNT INT;
BEGIN
    SELECT COUNT(*) INTO :V_COUNT
    FROM OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
    WHERE EVENT_ID = :P_EVENT_ID AND APPROVAL_STATUS = 'PENDING';
    IF (:V_COUNT = 0) THEN
        RETURN 'No pending fix found for event: ' || :P_EVENT_ID;
    END IF;
    UPDATE OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
    SET APPROVAL_STATUS = 'REJECTED',
        APPROVED_BY = CURRENT_USER(),
        APPROVED_AT = CURRENT_TIMESTAMP(),
        REJECTION_REASON = :P_REASON
    WHERE EVENT_ID = :P_EVENT_ID AND APPROVAL_STATUS = 'PENDING';
    RETURN 'REJECTED by ' || CURRENT_USER() || ' — Reason: ' || :P_REASON;
END;


CREATE OR REPLACE PROCEDURE OPS_AI_MONITOR.AI_ENGINE.SP_BULK_APPROVE(P_SEVERITY VARCHAR DEFAULT NULL)
RETURNS VARCHAR
LANGUAGE SQL
AS
DECLARE
    V_COUNT INT;
BEGIN
    IF (:P_SEVERITY IS NOT NULL) THEN
        UPDATE OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
        SET APPROVAL_STATUS = 'APPROVED',
            APPROVED_BY = CURRENT_USER(),
            APPROVED_AT = CURRENT_TIMESTAMP()
        WHERE APPROVAL_STATUS = 'PENDING' AND SEVERITY = :P_SEVERITY;
    ELSE
        UPDATE OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
        SET APPROVAL_STATUS = 'APPROVED',
            APPROVED_BY = CURRENT_USER(),
            APPROVED_AT = CURRENT_TIMESTAMP()
        WHERE APPROVAL_STATUS = 'PENDING';
    END IF;
    V_COUNT := SQLROWCOUNT;
    RETURN :V_COUNT || ' fixes approved by ' || CURRENT_USER();
END;


-- === HIL: VIEWS ===
CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_PENDING_APPROVALS AS
SELECT
    APPROVAL_ID, EVENT_ID, PIPELINE_TYPE, SEVERITY,
    ROUND(CONFIDENCE_SCORE * 100, 1) AS CONFIDENCE_PCT,
    LEFT(ROOT_CAUSE, 200) AS ROOT_CAUSE,
    LEFT(SUGGESTED_FIX, 200) AS SUGGESTED_FIX,
    FIX_SQL, APPROVAL_POLICY, CREATED_AT,
    DATEDIFF('MINUTE', CREATED_AT, CURRENT_TIMESTAMP()) AS WAITING_MINUTES
FROM OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
WHERE APPROVAL_STATUS = 'PENDING'
ORDER BY CASE SEVERITY WHEN 'HIGH' THEN 1 WHEN 'MEDIUM' THEN 2 ELSE 3 END, CREATED_AT ASC;


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_APPROVAL_STATS AS
SELECT
    APPROVAL_STATUS, SEVERITY, COUNT(*) AS CNT,
    AVG(CONFIDENCE_SCORE) AS AVG_CONFIDENCE,
    AVG(CASE WHEN APPROVED_AT IS NOT NULL THEN DATEDIFF('MINUTE', CREATED_AT, APPROVED_AT) END) AS AVG_APPROVAL_MINUTES
FROM OPS_AI_MONITOR.AI_ENGINE.FIX_APPROVAL_QUEUE
GROUP BY APPROVAL_STATUS, SEVERITY
ORDER BY APPROVAL_STATUS, SEVERITY;


-- === POST-FIX VALIDATION: TASK ===
CREATE OR REPLACE TASK OPS_AI_MONITOR.EVENT_LOGS.TSK_VALIDATE_FIXES
    WAREHOUSE = OPS_MONITOR_WH
    COMMENT = 'Post-fix validation: generates and runs health checks after AI fixes execute'
    AFTER OPS_AI_MONITOR.EVENT_LOGS.TSK_AUTO_FIX
AS
DECLARE
    C1 CURSOR FOR
        SELECT
            AFL.EVENT_ID,
            AFL.FIX_SQL,
            LEFT(FE.ERROR_MESSAGE, 500) AS ORIGINAL_ERROR,
            LEFT(COALESCE(FE.QUERY_TEXT, ''), 500) AS ORIGINAL_QUERY
        FROM OPS_AI_MONITOR.AI_ENGINE.AUTO_FIX_LOGS AFL
        JOIN OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE
            ON AFL.EVENT_ID = FE.EVENT_ID
        LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.POST_FIX_VALIDATION PFV
            ON AFL.EVENT_ID = PFV.EVENT_ID
        WHERE AFL.EXECUTION_STATUS = 'SUCCESS'
          AND PFV.EVENT_ID IS NULL
          AND AFL.EXECUTED_AT >= DATEADD('HOUR', -2, CURRENT_TIMESTAMP())
        ORDER BY AFL.EXECUTED_AT DESC
        LIMIT 5;
    V_EVENT_ID       VARCHAR;
    V_FIX_SQL        VARCHAR;
    V_ORIGINAL_ERROR VARCHAR;
    V_ORIGINAL_QUERY VARCHAR;
    V_TEST_QUERY     VARCHAR;
    V_TEST_RESULT    VARCHAR;
    V_STATUS         VARCHAR;
    V_VERDICT        VARCHAR;
BEGIN
    OPEN C1;
    FOR REC IN C1 DO
        V_EVENT_ID       := REC.EVENT_ID;
        V_FIX_SQL        := REC.FIX_SQL;
        V_ORIGINAL_ERROR := REC.ORIGINAL_ERROR;
        V_ORIGINAL_QUERY := REC.ORIGINAL_QUERY;
        V_TEST_QUERY     := NULL;
        V_TEST_RESULT    := NULL;
        V_STATUS         := 'PENDING';
        V_VERDICT        := NULL;
        BEGIN
            SELECT SNOWFLAKE.CORTEX.COMPLETE(
                'mistral-large2',
                CONCAT(
                    'You are a Snowflake SQL validation expert. Generate a single SELECT query to verify a fix was successful.',
                    ' The query must be safe (read-only SELECT only, no DML). Return ONLY the SQL, nothing else.',
                    ' If you cannot generate a validation query, return: SELECT 1 AS VALIDATION_OK;',
                    ' ORIGINAL ERROR: ', :V_ORIGINAL_ERROR,
                    ' FIX APPLIED: ', :V_FIX_SQL,
                    ' ORIGINAL QUERY: ', :V_ORIGINAL_QUERY
                )
            ) INTO :V_TEST_QUERY;
            V_TEST_QUERY := TRIM(:V_TEST_QUERY);
            IF (LEFT(:V_TEST_QUERY, 3) = '```') THEN
                V_TEST_QUERY := TRIM(REGEXP_REPLACE(:V_TEST_QUERY, '```[a-z]*', ''));
            END IF;
            IF (UPPER(LEFT(TRIM(:V_TEST_QUERY), 6)) != 'SELECT') THEN
                V_TEST_QUERY := 'SELECT 1 AS VALIDATION_OK';
            END IF;
            IF (UPPER(:V_TEST_QUERY) LIKE '%DROP %'
                OR UPPER(:V_TEST_QUERY) LIKE '%DELETE %'
                OR UPPER(:V_TEST_QUERY) LIKE '%INSERT %'
                OR UPPER(:V_TEST_QUERY) LIKE '%UPDATE %'
                OR UPPER(:V_TEST_QUERY) LIKE '%MERGE %'
                OR UPPER(:V_TEST_QUERY) LIKE '%CREATE %'
                OR UPPER(:V_TEST_QUERY) LIKE '%ALTER %'
                OR UPPER(:V_TEST_QUERY) LIKE '%GRANT %'
                OR UPPER(:V_TEST_QUERY) LIKE '%REVOKE %'
                OR UPPER(:V_TEST_QUERY) LIKE '%TRUNCATE %')
            THEN
                V_TEST_QUERY  := 'SELECT ''UNSAFE_QUERY_BLOCKED'' AS RESULT';
                V_TEST_RESULT := 'AI generated unsafe validation query - blocked';
                V_STATUS      := 'SKIPPED';
                V_VERDICT     := 'VALIDATION_BLOCKED';
            ELSE
                BEGIN
                    EXECUTE IMMEDIATE :V_TEST_QUERY;
                    V_TEST_RESULT := 'Query executed successfully';
                    V_STATUS      := 'SUCCESS';
                    V_VERDICT     := 'FIX_VERIFIED';
                EXCEPTION
                    WHEN OTHER THEN
                        V_TEST_RESULT := SQLERRM;
                        IF (POSITION(:V_ORIGINAL_ERROR IN SQLERRM) > 0
                            OR SQLERRM ILIKE '%does not exist%'
                            OR SQLERRM ILIKE '%not authorized%'
                            OR SQLERRM ILIKE '%compilation error%') THEN
                            V_STATUS  := 'FAILED';
                            V_VERDICT := 'FIX_FAILED';
                        ELSE
                            V_STATUS  := 'PARTIAL';
                            V_VERDICT := 'FIX_PARTIAL';
                        END IF;
                END;
            END IF;
        EXCEPTION
            WHEN OTHER THEN
                V_TEST_QUERY  := 'N/A - validation generation failed';
                V_TEST_RESULT := SQLERRM;
                V_STATUS      := 'ERROR';
                V_VERDICT     := 'VALIDATION_ERROR';
        END;
        INSERT INTO OPS_AI_MONITOR.AI_ENGINE.POST_FIX_VALIDATION
            (EVENT_ID, FIX_SQL, TEST_QUERY, TEST_RESULT, VALIDATION_STATUS, VALIDATION_VERDICT, ORIGINAL_ERROR, ORIGINAL_QUERY)
        VALUES
            (:V_EVENT_ID, :V_FIX_SQL, :V_TEST_QUERY, :V_TEST_RESULT, :V_STATUS, :V_VERDICT, :V_ORIGINAL_ERROR, :V_ORIGINAL_QUERY);
    END FOR;
    CLOSE C1;
END;

CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_VALIDATION_SUMMARY AS
SELECT
    PFV.VALIDATION_ID, PFV.EVENT_ID,
    FE.SOURCE_TYPE AS PIPELINE_TYPE, EA.SEVERITY,
    PFV.VALIDATION_STATUS, PFV.VALIDATION_VERDICT,
    PFV.FIX_SQL, PFV.TEST_QUERY, PFV.TEST_RESULT,
    LEFT(PFV.ORIGINAL_ERROR, 200) AS ORIGINAL_ERROR,
    PFV.VALIDATED_AT
FROM OPS_AI_MONITOR.AI_ENGINE.POST_FIX_VALIDATION PFV
LEFT JOIN OPS_AI_MONITOR.EVENT_LOGS.FAILURE_EVENTS FE ON PFV.EVENT_ID = FE.EVENT_ID
LEFT JOIN OPS_AI_MONITOR.AI_ENGINE.ERROR_ANALYSIS EA ON PFV.EVENT_ID = EA.EVENT_ID
ORDER BY PFV.VALIDATED_AT DESC;


CREATE OR REPLACE VIEW OPS_AI_MONITOR.AI_ENGINE.V_VALIDATION_METRICS AS
SELECT
    COUNT(*) AS TOTAL_VALIDATIONS,
    COUNT(CASE WHEN VALIDATION_STATUS = 'SUCCESS' THEN 1 END) AS VERIFIED_OK,
    COUNT(CASE WHEN VALIDATION_STATUS = 'PARTIAL' THEN 1 END) AS PARTIAL,
    COUNT(CASE WHEN VALIDATION_STATUS = 'FAILED' THEN 1 END) AS FAILED,
    COUNT(CASE WHEN VALIDATION_STATUS = 'SKIPPED' THEN 1 END) AS SKIPPED,
    ROUND(COUNT(CASE WHEN VALIDATION_STATUS = 'SUCCESS' THEN 1 END) * 100.0 / NULLIF(COUNT(*), 0), 1) AS SUCCESS_RATE_PCT
FROM OPS_AI_MONITOR.AI_ENGINE.POST_FIX_VALIDATION;
